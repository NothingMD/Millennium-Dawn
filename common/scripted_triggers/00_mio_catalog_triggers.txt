# Dispatches to the per-MIO `_unlock_btn_enabled` trigger by interpolating the current entry's
# token key. Used by both the click gate and the `[!]` tooltip — meta_trigger substitution keeps
# the per-MIO ✓/✗ checkmarks intact behind a single static reference.
mio_catalog_entry_prereqs_yes = {
	meta_trigger = {
		text = {
			[TRIG] = yes
		}
		TRIG = "[?global.mio_catalog_all_tokens^v.GetTokenKey]_unlock_btn_enabled"
	}
}
