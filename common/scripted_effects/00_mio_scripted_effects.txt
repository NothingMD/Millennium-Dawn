# Effect: free_trait_picks_used
# Purpose: Displays the number of free_trait_picks for a MIO to ensure there is a display and an notification to the player to make it easier to know when and why it's free
free_trait_picks_used = {
	custom_effect_tooltip = number_of_mio_free_trait_picks
	add_to_variable = { free_trait_picks = -1 }

	if = { limit = { check_variable = { free_trait_picks = 0 } }
		clear_variable = free_trait_picks
	}
}

expenditure_for_mio_upgrade = {
	if = { limit = { check_variable = { free_trait_picks > 0 } }
		free_trait_picks_used = yes
	}
	else = {
		FROM = {
			small_expenditure = yes
		}
	}
}
